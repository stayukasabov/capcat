# Capcat Video Ad — Implementation Plan

> **For Claude:** REQUIRED SUB-SKILL: Use superpowers:executing-plans to implement this plan task-by-task.

**Goal:** Build a ~110s Remotion video ad with 40 individually-editable beat components, driven by real audio narration, using the Capcat design system.

**Architecture:** Single `CapcatAd` composition. `<Audio>` plays narration globally. `<Series>` sequences all 40 `<Series.Sequence>` beats consecutively. Each beat is an isolated `S{N}_B{M}.tsx` file — illustration + text, no inter-beat dependencies. Timing comes from `beatTimings.ts` generated by audio analysis.

**Tech Stack:** Remotion 4, React 18, TypeScript, `@remotion/media` (Audio), `@remotion/google-fonts/SourceSerif4`, `lucide-react`, Python `librosa` (pre-build audio analysis)

**Design doc:** `docs/plans/2026-03-08-capcat-ad-design.md`

**Remotion rules (enforced throughout):**
- All animation via `useCurrentFrame()` + `interpolate()` + `spring()` — NO CSS transitions
- All assets via `staticFile()` — NO direct paths
- Every `<Sequence>` uses `premountFor={fps}`
- Typewriter: string slicing only

**Step 1: Create the analysis script**

```python
# Video/scripts/analyze_audio.py
import librosa
import json
import sys

def analyze(path):
    y, sr = librosa.load(path, sr=None)
    # Detect onsets (transients — narration phrase starts + swoosh accents)
    onset_times = librosa.onset.onset_detect(y=y, sr=sr, units='time',
                                              hop_length=512, backtrack=True)
    print(json.dumps([round(float(t), 3) for t in onset_times], indent=2))

analyze(sys.argv[1])
```

**Step 3: Create `Typewriter.tsx`**

```tsx
// Video/src/components/Typewriter.tsx
import React from "react";
import { interpolate, useCurrentFrame } from "remotion";
import { colors } from "../tokens";

const CHARS_PER_FRAME_NORMAL = 4;   // ≤5 words, 52px
const CHARS_PER_FRAME_LONG = 5;     // >5 words, 38px

export const Typewriter: React.FC<{
  text: string;
  startFrame?: number;
  charsPerFrame?: number;
  showCursor?: boolean;
}> = ({ text, startFrame = 4, charsPerFrame = CHARS_PER_FRAME_NORMAL, showCursor = true }) => {
  const frame = useCurrentFrame();
  const elapsed = Math.max(0, frame - startFrame);
  const chars = Math.min(text.length, Math.floor(elapsed * charsPerFrame));
  const displayed = text.slice(0, chars);
  const isTyping = chars < text.length;

  const cursorOpacity = interpolate(
    frame % 20,
    [0, 9, 10, 19],
    [1, 1, 0, 0],
    { extrapolateRight: "clamp" }
  );

  return (
    <span>
      {displayed}
      {showCursor && isTyping && (
        <span style={{ opacity: cursorOpacity, color: colors.accentPrimary }}>▌</span>
      )}
    </span>
  );
};
```

**Step 4: Create `BeatWrapper.tsx`**

Standard shell every beat uses for fade-in and fade-out.

```tsx
// Video/src/components/BeatWrapper.tsx
import React from "react";
import { AbsoluteFill, interpolate, useCurrentFrame, useVideoConfig } from "remotion";
import { colors } from "../tokens";

export const BeatWrapper: React.FC<{
  children: React.ReactNode;
  background?: string;
}> = ({ children, background = colors.paper }) => {
  const frame = useCurrentFrame();
  const { durationInFrames } = useVideoConfig();

  const fadeIn = interpolate(frame, [0, 8], [0, 1], {
    extrapolateLeft: "clamp",
    extrapolateRight: "clamp",
  });

  const fadeOut = interpolate(
    frame,
    [durationInFrames - 10, durationInFrames],
    [1, 0],
    { extrapolateLeft: "clamp", extrapolateRight: "clamp" }
  );

  const opacity = Math.min(fadeIn, fadeOut);

  return (
    <AbsoluteFill style={{ backgroundColor: background, opacity }}>
      {children}
    </AbsoluteFill>
  );
};
```

```tsx
  "Catch articles from a bundle of sources",
  "Catch articles from a list of sources",
  "Catch from a single source",
  "Catch a single article by URL",
  "Manage Sources (add/remove/configure)",
  "Exit",
];

const FRAMES_PER_STEP = 12;
const MENU_START_FRAME = 40;

export const TUIDisplay: React.FC = () => {
  const frame = useCurrentFrame();

  // Logo fade in
  const logoOpacity = interpolate(frame, [0, 15], [0, 1], {
    extrapolateLeft: "clamp", extrapolateRight: "clamp",
  });

  // Menu lines appear 3 frames apart after logo
  const menuStart = 20;
  const menuLineOpacity = (i: number) =>
    interpolate(frame, [menuStart + i * 3, menuStart + i * 3 + 6], [0, 1], {
      extrapolateLeft: "clamp", extrapolateRight: "clamp",
    });

  // Pointer position cycles through options
  const pointerIndex = (() => {
    if (frame < MENU_START_FRAME) return 0;
    const step = Math.floor((frame - MENU_START_FRAME) / FRAMES_PER_STEP);
    // Sequence: 0→1→2→3→4→5→4→3→2→1→0, then hold at 0

    return sequence[Math.min(step, sequence.length - 1)];
  })();

  return (
    <div
      style={{
        backgroundColor: "#1a1212",
        borderRadius: 10,
        padding: "20px 28px",
        fontFamily: fonts.mono,
        fontSize: 18,
```

```tsx
// Video/src/components/ProgressDisplay.tsx
import React from "react";
import { interpolate, useCurrentFrame, useVideoConfig } from "remotion";
import { colors, fonts } from "../tokens";

const SPINNER_CHARS = ["◒", "◓", "◔", "◕"];
const TOTAL_ITEMS = 27;
const DOT_COUNT = 6;

function getStatus(pct: number): string {
  if (pct < 0.15) return "STARTING";
  if (pct < 0.55) return "FETCHING";
  if (pct < 0.9) return "CONVERTING";
  return "COMPLETE";
}

export const ProgressDisplay: React.FC = () => {
  const frame = useCurrentFrame();
  const { durationInFrames } = useVideoConfig();

  const progress = interpolate(frame, [10, durationInFrames - 10], [0, 1], {
    extrapolateLeft: "clamp", extrapolateRight: "clamp",
  });

  const spinnerIndex = Math.floor(frame / 8) % SPINNER_CHARS.length;
  const spinner = SPINNER_CHARS[spinnerIndex];

  const currentItem = Math.floor(progress * TOTAL_ITEMS);
  const pct = (currentItem / TOTAL_ITEMS) * 100;
  const filledDots = Math.floor(progress * DOT_COUNT);
  const dots = Array.from({ length: DOT_COUNT }, (_, i) => i < filledDots ? "◉" : "◎").join(" ");
  const status = getStatus(progress);

  return (
    <div
      style={{
        backgroundColor: "#1a1212",
        borderRadius: 10,
        padding: "20px 28px",
        fontFamily: fonts.mono,
        fontSize: 22,
        color: colors.accentPrimary,
        minWidth: 640,
        boxShadow: "0 4px 32px rgba(0,0,0,0.5)",
      }}
    >
      {`${spinner} CATCHING ▶  NATURE ${dots}  ${currentItem}/${TOTAL_ITEMS} (${pct.toFixed(1)}%) (${status})`}
    </div>
  );
};
```

## Task 9: Scene 5 Beats (S5_B1 – S5_B3)

**Files:** Create `Video/src/beats/s5/`

**S5_B2** — cat entrance with blur + elastic (critical beat):

```tsx
// Video/src/beats/s5/S5_B2.tsx
import React from "react";
import { Img, interpolate, spring, staticFile, useCurrentFrame, useVideoConfig } from "remotion";
import { BeatWrapper } from "../../components/BeatWrapper";
import { IllustrationZone } from "../../components/IllustrationZone";
import { TextZone, getFontSize } from "../../components/TextZone";
import { Typewriter } from "../../components/Typewriter";

const TEXT = "Capcat";

export const S5_B2: React.FC<{ fontFamily: string }> = ({ fontFamily }) => {
  const frame = useCurrentFrame();
  const { fps } = useVideoConfig();

  // Elastic entrance from left — bouncy spring
  const catSpring = spring({ frame, fps, config: { damping: 8 }, durationInFrames: 40 });
  const catX = interpolate(catSpring, [0, 1], [-600, 0]);

  // Motion blur: high blur at start, 0 at rest
  const blur = interpolate(catSpring, [0, 0.6, 1], [12, 3, 0], {
    extrapolateLeft: "clamp", extrapolateRight: "clamp",
  });

  return (
    <BeatWrapper>
      <IllustrationZone>
        <div
          style={{
            transform: `translateX(${catX}px)`,
            filter: `blur(${blur}px)`,
          }}
        >
          <Img src={staticFile("ILLUSTRATION/Capcat-Cat-Color.svg")} style={{ width: 400, height: 400 }} />
        </div>
      </IllustrationZone>
      <TextZone fontFamily={fontFamily} fontSize={getFontSize(TEXT)}>
        <Typewriter text={TEXT} />
      </TextZone>
    </BeatWrapper>
  );
};
```

**S5_B3** — crowd behind cat. Cat stays. Crowd comes from below:

## Editing Guide (Post-Build)

| Goal | Action |
|------|--------|
| Change beat duration | Edit `durationSec` in `beatTimings.ts` for that `id` |
| Change illustration | Edit `Video/src/beats/s{N}/S{N}_B{M}.tsx` illustration zone |
| Change text | Edit `TEXT` constant in `Video/src/beats/s{N}/S{N}_B{M}.tsx` |
| Change animation spring | Edit `spring()` config in the specific beat file |
| Change font size rule | Edit `getFontSize()` in `TextZone.tsx` |
| Change global colors | Edit `tokens.ts` |
| Re-run audio analysis | Run `python scripts/analyze_audio.py`, update `beatTimings.ts` |
