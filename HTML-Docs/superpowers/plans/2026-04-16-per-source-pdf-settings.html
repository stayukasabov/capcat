Per-Source PDF Settings Implementation Plan


> For agentic workers: REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (- [ ]) syntax for tracking.


Goal: Allow each source YAML to declare its own PDF download preference and size limit, and add a "Source defaults" third option to the TUI PDF prompt using an arrow-key menu.
Architecture: Three thin additions to existing infrastructure: (1) replace the TUI questionary.confirm PDF prompt with questionary.select + info block + "Source defaults" option; (2) add _source_max_pdf_bytes() helper to NewsSourceArticleFetcher and replace the two hardcoded get_config().pdf.max_pdf_size_bytes call sites; (3) add media: blocks to six builtin source YAML configs. No structural changes — _resolve_media and media_overrides already handle the boolean; only the size cap is new.
Tech Stack: Python, questionary (already imported in interactive.py), source YAML configs, article_fetcher.py, CLAUDE.md.
Branch: feat/per-source-pdf-settings

def _source_max_pdf_bytes(self) -> int:
    """Return the PDF size limit in bytes for this source.

    Checks source config YAML for media.max_pdf_size_mb first;
    falls back to the global pdf.max_pdf_size_bytes.
    """
    mb = self.source_config.get("media", {}).get("max_pdf_size_mb")
    if mb is not None:
        return int(mb) * 1024 * 1024
    return get_config().pdf.max_pdf_size_bytes

class TestTuiFlagEmission:
    """TUI PDF menu emits correct flags for each choice."""

    def _run_confirm_and_execute(self, pdf_choice, action="bundle", selection="all"):
        """Invoke _confirm_and_execute with a mocked PDF questionary answer."""
        import capcat.core.interactive as interactive_mod

        captured_args = []

        def fake_dispatch(args):
            captured_args.extend(args)

        with (
            patch.object(interactive_mod.questionary, "select") as mock_select,
            patch("capcat.core.interactive._dispatch", side_effect=fake_dispatch),
            patch("capcat.core.interactive._show_completion_screen"),
            patch("capcat.core.interactive.is_tui_active", return_value=False),
        ):
            # First select call is the PDF menu; return the desired choice
            mock_select.return_value.ask.return_value = pdf_choice
            interactive_mod._confirm_and_execute(action, selection, generate_html=False)

        return captured_args

    def test_source_defaults_emits_no_pdf_flag(self):
        """'Source defaults' must not append --pdfs or --no-pdfs."""
        args = self._run_confirm_and_execute("source_defaults")
        assert "--pdfs" not in args
        assert "--no-pdfs" not in args

    def test_yes_emits_pdfs_flag(self):
        args = self._run_confirm_and_execute("yes")
        assert "--pdfs" in args
        assert "--no-pdfs" not in args

    def test_no_emits_no_pdfs_flag(self):
        args = self._run_confirm_and_execute("no")
        assert "--no-pdfs" in args
        assert "--pdfs" not in args
